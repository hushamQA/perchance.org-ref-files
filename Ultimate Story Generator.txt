  // Helper: Convert hex to rgba for focus color
  function hexToRgba(hex, alpha) {
    let c;
    if(/^#([A-Fa-f0-9]{3}){1,2}$/.test(hex)){
      c= hex.substring(1).split('');
      if(c.length== 3){
        c= [c[0], c[0], c[1], c[1], c[2], c[2]];
      }
      c= '0x'+c.join('');
      return 'rgba('+[(c>>16)&255, (c>>8)&255, c&255].join(',')+','+alpha+')';
    }
    return hex; // fallback
  }

  // Helper: Update theme reactive colors (Backgrounds, Inputs, Borders)
  function updateThemeColors() {
    let hasCustomBg = document.body.classList.contains("has-custom-bg");

    if (hasCustomBg) {
      // Custom BG Mode
      // Make body background transparent so the image shows
      document.body.style.setProperty('--bg-body', 'transparent');
      
      // Panels/inputs stay opaque via --bg-panel which is not affected
      document.body.classList.add('has-custom-bg');
    } else {
      // No Custom BG Mode
      // Remove all overrides to use the active background theme
      document.body.style.removeProperty('--bg-body');
      document.body.classList.remove('has-custom-bg');
    }
  }

      await kv.storyGenerator.set("mainState", state);
    } catch (e) {
      console.error("Failed to save state:", e);
    }
  }

  // Load state from persistent storage
  async function loadState() {
    try {
      let state = await kv.storyGenerator.get("mainState");
      if (state) {
        characters = state.characters || [];
        settings = state.settings || [];
        chapterSummaries = state.chapterSummaries || [];
        currentStoryText = state.currentStoryText || "";
        
        if (state.plotAndIdeas && !state.plot) {
          plot = state.plotAndIdeas;
        } else {
          plot = state.plot || "";
        }
        
        themesStr = state.themes || "";
        ideas = state.ideas || "";
        negativePrompts = state.negativePrompts || "";

        voice = state.voice || "Default";
        bannedWords = state.bannedWords || "";
        customSystemPrompt = state.customSystemPrompt || "";

        // UI Customization - Handle Button Themes (Old & New format)
        if (state.customAccentColor && !state.selectedTheme && !state.customButtonColors) {
          // Old format: convert to default theme with custom color
          selectedTheme = "default";
          themes.default.accent = state.customAccentColor;
          themes.default.accentHover = adjustColorBrightness(state.customAccentColor, -20);
        } else {
          selectedTheme = state.selectedTheme || "default";
        }
        
        // Migration for Custom Button Colors
        if (state.customButtonColors) {
          customButtonColors = state.customButtonColors;
        } else if (state.customButtonColor) {
          customButtonColors = { accent: state.customButtonColor, success: "", danger: "", warning: "" };
        } else {
          customButtonColors = { accent: "", success: "", danger: "", warning: "" };
        }

        // UI Customization - Handle Background Themes
        if (state.selectedBackgroundTheme) {
          selectedBackgroundTheme = state.selectedBackgroundTheme;
        } else if (state.isLightMode) {
          // Migrate old light mode toggle to new background theme system
          selectedBackgroundTheme = "light";
        } else {
          selectedBackgroundTheme = "dark";
        }

        // Migration for Custom Background Colors
        if (state.customBgColors) {
          customBgColors = state.customBgColors;
        } else if (state.customBgColor) {
          customBgColors = { body: state.customBgColor, panel: "", input: "" };
        } else {
          customBgColors = { body: "", panel: "", input: "" };
        }

        // UI Customization - Handle Font Family (separate from colors now)
        selectedFontFamily = state.selectedFontFamily || "default";
        
        // Migration for custom font colors
        customFontMainColor = state.customFontMainColor || "#f0f0f0";
        customFontMutedColor = state.customFontMutedColor || "#a0a0a0";

          customFontMainColor = state.customFontColor;
        }
        
        customBackgroundImage = state.customBackgroundImage || "";
        
        if (typeof state.currentLengthOption === 'string') {
          if (state.currentLengthOption.includes('unlimited')) {
            currentLengthOption = { val: 3, unit: 'paragraphs', isUnlimited: true };
          } else {
            let parts = state.currentLengthOption.split(' ');
            let val = parseInt(parts[0]) || 3;
            let unit = parts.length > 1 ? parts[1].replace(/s$/, '') : 'paragraphs';
            if (unit !== 'paragraphs' && unit !== 'sentences') unit = 'paragraphs';
            currentLengthOption = { val: val, unit: unit, isUnlimited: false };
          }
        } else {
          currentLengthOption = state.currentLengthOption || { val: 3, unit: 'paragraphs', isUnlimited: false };
        }

        charGenDraft = state.charGenDraft || "";
        assistantDraft = state.assistantDraft || "";
        
        if (state.collapsedSections !== undefined) {
          if (state.collapsedSections.plot !== undefined || state.collapsedSections.negativePrompts !== undefined) {
            collapsedSections.loreAndPrompting = !(state.collapsedSections.plot === false || state.collapsedSections.negativePrompts === false);
          } else {
            collapsedSections.loreAndPrompting = state.collapsedSections.loreAndPrompting !== undefined ? state.collapsedSections.loreAndPrompting : true;
          }
          collapsedSections.characters = state.collapsedSections.characters !== undefined ? state.collapsedSections.characters : true;
          collapsedSections.settings = state.collapsedSections.settings !== undefined ? state.collapsedSections.settings : true;
          collapsedSections.summaries = state.collapsedSections.summaries !== undefined ? state.collapsedSections.summaries : true;
        }

        hasSeenTutorial = state.hasSeenTutorial || false;

        storyOutputEl.value = currentStoryText;
        plotInputEl.value = plot;
        themesInputEl.value = themesStr;
        ideasInputEl.value = ideas;
        negativePromptsInputEl.value = negativePrompts;

        bannedWordsInputEl.value = bannedWords;
        systemPromptInputEl.value = customSystemPrompt;
        
        lengthNumberInput.value = currentLengthOption.val;
        lengthUnitInput.value = currentLengthOption.unit;
        lengthUnlimitedCheck.checked = currentLengthOption.isUnlimited;
        
        charGenOutputEl.value = charGenDraft;
        assistantOutputEl.value = assistantDraft;

        if (selectedTheme === 'custom' && (customButtonColors.accent || customButtonColors.success || customButtonColors.danger || customButtonColors.warning)) {
          applyCustomButtonTheme();
        } else {
          applyTheme(selectedTheme);
        }

        // Apply Background Theme
        if (selectedBackgroundTheme === 'custom' && (customBgColors.body || customBgColors.panel || customBgColors.text)) {
          applyCustomBackgroundTheme();
        } else {
          applyBackgroundTheme(selectedBackgroundTheme);
        }

        // Apply Font Settings (Family and Colors)
        applyFontSettings();

        if (customBackgroundImage) {
          document.body.style.backgroundImage = `url(${customBackgroundImage})`;
          document.body.classList.add('has-custom-bg');
          bgPreviewImg.src = customBackgroundImage;
          bgPreviewContainer.style.display = 'block';
          removeBgBtn.hidden = false;

        }

        renderFontFamilyDropdown();
        updateFontColorInputs();

        renderFinishedChapters();
        restoreCollapsedState();

        if (!hasSeenTutorial) {
          openHowItWorks();
        }
      }
    } catch (e) {
      console.error("Failed to load state:", e);
    }
  }

  // Helper: Simple color brightness adjustment for hover state
  function adjustColorBrightness(hex, percent) {
    let num = parseInt(hex.replace("#", ""), 16),
    amt = Math.round(2.55 * percent),
    R = (num >> 16) & 255,
    B = ((num >> 8) & 0x00FF) + amt,
    G = (num & 0x0000FF) + amt;
    return "#" + (0x1000000 + (R<255?R<1?0:R:255)*0x10000 + (B<255?B<1?0:B:255)*0x100 + (G<255?G<1?0:G:255)).toString(16).slice(1);
  }

  // Helper: Hex to RGB object
  function hexToRgb(hex) {
    let result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
    return result ? {
      r: parseInt(result[1], 16),
      g: parseInt(result[2], 16),
      b: parseInt(result[3], 16)
    } : null;
  }

  // Function to update random glows for all elements
  function updateRandomGlows() {
    let themeColors = [];
    
    // Gather colors based on active theme
    if (selectedTheme === 'custom') {
      if (customButtonColors.accent) themeColors.push(customButtonColors.accent);
      if (customButtonColors.success) themeColors.push(customButtonColors.success);
      if (customButtonColors.danger) themeColors.push(customButtonColors.danger);
      if (customButtonColors.warning) themeColors.push(customButtonColors.warning);
      // Fallbacks if empty
      if (themeColors.length === 0) themeColors = ['#6366f1', '#10b981', '#ef4444', '#f59e0b'];
    } else {
      let theme = themes[selectedTheme];
      if (theme) {
        themeColors.push(theme.accent, theme.success, theme.danger, theme.warning);
      } else {
        themeColors = ['#6366f1', '#10b981', '#ef4444', '#f59e0b'];
      }
    }

    // Target elements: Buttons, Checkboxes, Radio buttons
    // We specifically target buttons and inputs to distribute the colors
    let elements = document.querySelectorAll('button, input[type="checkbox"], input[type="radio"]');
    
    elements.forEach(el => {
      if (themeColors.length > 0) {
        let randomHex = themeColors[Math.floor(Math.random() * themeColors.length)];
        let randomRgba = hexToRgba(randomHex, 0.4);
        
        // Set glow color (for borders and shadows)
        el.style.setProperty('--glow-color', randomRgba);

        // If it's a button, assign the random color to the background override vars
        if (el.tagName === 'BUTTON') {
            let hoverHex = adjustColorBrightness(randomHex, -20);
            el.style.setProperty('--btn-bg-override', randomHex);
            el.style.setProperty('--btn-hover-override', hoverHex);
        }
        
        // If it's a checkbox/radio, set the accent color directly
        if (el.type === 'checkbox' || el.type === 'radio') {
            el.style.accentColor = randomHex;
        }
      }
    });
  }

  // Helper: Apply button theme colors
  function applyTheme(themeKey) {
    let theme = themes[themeKey];
    if (!theme) return;

    document.documentElement.style.setProperty('--accent-primary', theme.accent);
    document.documentElement.style.setProperty('--accent-primary-hover', theme.accentHover);
    document.documentElement.style.setProperty('--accent-success', theme.success);
    document.documentElement.style.setProperty('--accent-success-hover', theme.successHover);
    document.documentElement.style.setProperty('--accent-danger', theme.danger);
    document.documentElement.style.setProperty('--accent-danger-hover', theme.dangerHover);
    document.documentElement.style.setProperty('--accent-warning', theme.warning);
    document.documentElement.style.setProperty('--accent-warning-hover', theme.warningHover);
    
    // Set focus color based on accent color with 40% opacity
    document.documentElement.style.setProperty('--focus-color', hexToRgba(theme.accent, 0.4));
    
    // Set button glow colors to match theme
    document.documentElement.style.setProperty('--btn-glow-primary', hexToRgba(theme.accent, 0.4));
    document.documentElement.style.setProperty('--btn-glow-success', hexToRgba(theme.success, 0.4));
    document.documentElement.style.setProperty('--btn-glow-danger', hexToRgba(theme.danger, 0.4));
    
    selectedTheme = themeKey;
    renderThemeGrid();
    updateRandomGlows();
  }

  function applyBackgroundTheme(themeKey) {
    let theme = backgroundThemes[themeKey];
    if (!theme) return;

    document.documentElement.style.setProperty('--bg-body', theme.bgBody);
    document.documentElement.style.setProperty('--bg-panel', theme.bgPanel);
    document.documentElement.style.setProperty('--bg-input', theme.bgInput);
    document.documentElement.style.setProperty('--bg-hover', theme.bgHover);
    document.documentElement.style.setProperty('--border-color', theme.borderColor);
    document.documentElement.style.setProperty('--border-light', theme.borderLight);
    document.documentElement.style.setProperty('--shadow-sm', theme.shadowSm);
    document.documentElement.style.setProperty('--shadow-md', theme.shadowMd);
    document.documentElement.style.setProperty('--shadow-lg', theme.shadowLg);

    // Sync light-mode class for specific CSS fallbacks if needed (e.g. for image opacity)
    if (theme.isLight) {
      document.body.classList.add('light-mode');
    } else {
      document.body.classList.remove('light-mode');

    updateThemeColors(); // Check custom bg opacity
    renderBackgroundThemeGrid();
  }

  // Apply Custom Button Theme (Multi-color)
  function applyCustomButtonTheme() {
    // Default colors
    let accent = customButtonColors.accent || "#6366f1";
    let success = customButtonColors.success || "#10b981";
    let danger = customButtonColors.danger || "#ef4444";
    let warning = customButtonColors.warning || "#f59e0b";

    // Accent
    document.documentElement.style.setProperty('--accent-primary', accent);
    // Hover (darker)
    let accentHover = adjustColorBrightness(accent, -20);
    document.documentElement.style.setProperty('--accent-primary-hover', accentHover);
    // Focus color
    document.documentElement.style.setProperty('--focus-color', hexToRgba(accent, 0.4));
    
    // Success
    document.documentElement.style.setProperty('--accent-success', success);
    let successHover = adjustColorBrightness(success, -20);
    document.documentElement.style.setProperty('--accent-success-hover', successHover);
    
    // Danger
    document.documentElement.style.setProperty('--accent-danger', danger);
    let dangerHover = adjustColorBrightness(danger, -20);
    document.documentElement.style.setProperty('--accent-danger-hover', dangerHover);
    
    // Warning
    document.documentElement.style.setProperty('--accent-warning', warning);
    let warningHover = adjustColorBrightness(warning, -20);
    document.documentElement.style.setProperty('--accent-warning-hover', warningHover);
    
    // Set button glow colors to match theme
    document.documentElement.style.setProperty('--btn-glow-primary', hexToRgba(accent, 0.4));
    document.documentElement.style.setProperty('--btn-glow-success', hexToRgba(success, 0.4));
    document.documentElement.style.setProperty('--btn-glow-danger', hexToRgba(danger, 0.4));
    
    renderThemeGrid();
    updateRandomGlows();
  }

  // Apply Custom Background Theme (Multi-color)
  function applyCustomBackgroundTheme() {
    // Default fallbacks
    let cBody = customBgColors.body || "#0f0f0f";
    let cPanel = customBgColors.panel || "#1a1a1a";
    let cInput = customBgColors.input || "#242424";
    
    // Determine light/dark mode based on body color luminance
    let rgbBody = hexToRgb(cBody);
    let isLight = rgbBody ? (0.299 * rgbBody.r + 0.587 * rgbBody.g + 0.114 * rgbBody.b > 128) : false;

    // Derive palette based on panel color (for hover and borders)
    let shift1, shift2, shift3;
    if (isLight) {
      shift1 = -10; shift2 = -20; shift3 = -30; // Darken
    } else {
      shift1 = 10; shift2 = 20; shift3 = 30;   // Lighten
    }

    let cHover = adjustColorBrightness(cPanel, shift1);
    
    let cBorder = isLight ? adjustColorBrightness(cPanel, -30) : adjustColorBrightness(cPanel, 40);
    let cBorderLight = isLight ? adjustColorBrightness(cPanel, -20) : adjustColorBrightness(cPanel, 20);

    let cShadow = isLight ? "rgba(0, 0, 0, 0.1)" : "rgba(0, 0, 0, 0.5)";

    // Apply Vars
    document.documentElement.style.setProperty('--bg-body', cBody);
    document.documentElement.style.setProperty('--bg-panel', cPanel);
    document.documentElement.style.setProperty('--bg-input', cInput);
    document.documentElement.style.setProperty('--bg-hover', cHover);
    document.documentElement.style.setProperty('--border-color', cBorder);
    document.documentElement.style.setProperty('--border-light', cBorderLight);
    document.documentElement.style.setProperty('--shadow-sm', cShadow === "rgba(0, 0, 0, 0.1)" ? "0 1px 2px rgba(0, 0, 0, 0.1)" : "0 1px 2px rgba(0, 0, 0, 0.3)");
    document.documentElement.style.setProperty('--shadow-md', cShadow === "rgba(0, 0, 0, 0.1)" ? "0 4px 6px rgba(0, 0, 0, 0.1)" : "0 4px 6px rgba(0, 0, 0, 0.4)");
    document.documentElement.style.setProperty('--shadow-lg', cShadow === "rgba(0, 0, 0, 0.1)" ? "0 10px 15px rgba(0, 0, 0, 0.1)" : "0 10px 15px rgba(0, 0, 0, 0.5)");

    // Handle body class for transparency logic
    if (isLight) document.body.classList.add('light-mode');
    else document.body.classList.remove('light-mode');
    
    updateThemeColors();
    renderBackgroundThemeGrid();
  }

  window.toggleSection = function(contentId, headerEl) {
    let contentEl = document.getElementById(contentId);
    let isExpanded = contentEl.classList.contains("visible");
    
    if (isExpanded) {

      collapsedSections.loreAndPrompting = !isExpanded;

      collapsedSections.characters = !isExpanded;
    } else if (contentId === "settingsContent") {
      collapsedSections.settings = !isExpanded;
    } else if (contentId === "summariesContent") {
      collapsedSections.summaries = !isExpanded;
    }
    
    saveState();
  };
  
  function restoreCollapsedState() {
    const sections = [
      { id: "loreAndPromptingContent", state: collapsedSections.loreAndPrompting },
      { id: "charactersContent", state: collapsedSections.characters },
      { id: "settingsContent", state: collapsedSections.settings },
      { id: "summariesContent", state: collapsedSections.summaries }
    ];
    
    sections.forEach(section => {
      let contentEl = document.getElementById(section.id);
      let headerEl = contentEl.previousElementSibling;
      
      if (section.state) {
        contentEl.classList.remove("visible");
        headerEl.classList.remove("expanded");

  function buildMemoryContext() {
    let context = "";
    if (plot) {
      context += `\n\nBackground/History:\n${plot}`;
    }
    if (themesStr) {
      context += `\n\nThemes:\n${themesStr}`;
    }
    if (ideas) {
      context += `\n\nStory Ideas:\n${ideas}`;
    }
    if (chapterSummaries.length > 0) {
      context += "\n\nPrevious Chapter Summaries (for context and continuity):\n";
      chapterSummaries.forEach(item => {
        context += `- ${item.title}: ${item.summary}\n`;
      });
    }
    if (characters.length > 0) {
      context += "\nCharacters:\n";
      characters.forEach(c => {
        context += `- ${c.name}: ${c.description}\n`;
      });
    }
    if (settings.length > 0) {
      context += "\nSettings:\n";
      settings.forEach(s => {
        context += `- ${s.name}: ${s.description}\n`;
      });
    }
    return context;
  }

  async function generate() {
    if (generationTask) {
      generationTask.stop();
      generationTask = null;
      generateBtn.textContent = "✨ Generate";
      loadingEl.hidden = true;
      return;
    }

    undoStack.push(storyOutputEl.value);
    if(undoStack.length > 20) undoStack.shift();

    let topic = topicInput.value.trim();
    if (!topic) topic = "an unexpected adventure";

    generateBtn.textContent = "⏹ Stop"; 
    loadingEl.hidden = false;

    try {
      let prompt = "";
      
      if (customSystemPrompt) {
        prompt += `${customSystemPrompt}\n\n`;
      }

      if (voice && voice !== "Default") {
        prompt += `VOICE/TONE STYLE: Write in a ${voice} voice.\n\n`;
      }

      let lengthInstruction = "";
      if (currentLengthOption.isUnlimited) {
        lengthInstruction = "Write an unlimited amount of content.";
      } else {
        let val = currentLengthOption.val > 0 ? currentLengthOption.val : 1;
        let unit = currentLengthOption.unit;
        let singularUnit = unit.endsWith('s') ? unit.slice(0, -1) : unit;
        lengthInstruction = `CRITICAL LENGTH CONSTRAINT: You MUST write STRICTLY ${val} ${unit}. You MUST stop immediately after the last ${singularUnit}. DO NOT write ${val + 1} ${unit}. Do not write any extra text, concluding remarks, or transitions beyond the ${val} ${unit} limit.`;
      }

      let isContinuation = currentStoryText.trim().length > 0;

      if (isContinuation) {
        currentStoryText += "\n\n";
        storyOutputEl.value = currentStoryText;
      }

      let memoryContext = buildMemoryContext();
      
      let negativeConstraints = "";
      let negativeItems = [];
      if (negativePrompts) negativeItems.push(`themes: ${negativePrompts}`);
      if (bannedWords) negativeItems.push(`words/phrases: ${bannedWords}`);
      
      if (negativeItems.length > 0) {
        negativeConstraints = `\n\nSTRICT NEGATIVE CONSTRAINTS:\nThe story must strictly AVOID the following: ${negativeItems.join(", ")}\nDo not include these themes, actions, words, or elements.`;
      }

      let antiRepetitionInstruction = "\n\nIMPORTANT: Avoid being repetitive with phrases and sentence structures. Keep the writing fresh and varied.";

      if (isContinuation) {
        let bridgeInstruction = `
          SMOOTH SCENE TRANSITION REQUIREMENT:
          The user is asking for a new event or scene based on: ("${topic}").
          DO NOT simply jump immediately to this new scene.
          Instead, first spend 1-3 sentences wrapping up the current moment/scene described in the story so far.
          Then, naturally transition to the new request ("${topic}") using a logical bridge (e.g., "Meanwhile," "Later that day," "After a short drive," etc.).
          The transition should feel continuous and fluid, avoiding sudden jumps in time or location without a connector.
        `;

        let detailedCharacterDescriptionInstruction = `
          IMPORTANT CHARACTER DESCRIPTION REQUIREMENT:
          If a new character is introduced, make sure to give them a thorough physical description. 
          This should include: detailed appearance (hair, eyes, body type, distinctive features), clothing, mannerisms, and any unique traits that make them memorable.
          The description should be comprehensive enough for readers to clearly visualize the character.
          Do not introduce a character without providing these details, even if it takes several sentences.
        `;

        prompt += `Continue the story based on the following instruction: "${topic}". ${lengthInstruction} Maintain the tone and style of the existing story.

The memory context below is provided for background information and loose inspiration. Use it only as a SUBTLE NUDGE to inform your writing style and general atmosphere. Do NOT let the memory context directly dictate the plot or content. Let your creativity lead primarily, using the memory only as faint background context; HOWEVER do not contradict anything that already exists in the memory context provided. ${memoryContext}

${negativeConstraints}

${antiRepetitionInstruction}

${bridgeInstruction}

${detailedCharacterDescriptionInstruction}

Story so far:
"""${currentStoryText}"""`;
      } else {
        prompt += `Write the **beginning** of a larger, multi-chapter story based on the following instruction: "${topic}". ${lengthInstruction} This is the opening scene or section, not a complete standalone short story. Focus on setting the scene, introducing key elements, and hooking the reader for the ongoing narrative.

The memory context below is provided for background information and loose inspiration. Use it only as a SUBTLE NUDGE to inform your writing style and general atmosphere. Do NOT let the memory context directly dictate the plot or content. Let your creativity lead primarily, using the memory only as faint background context; HOWEVER do not contradict anything that already exists in the memory context provided. ${memoryContext}

${negativeConstraints}

${antiRepetitionInstruction}

IMPORTANT CHARACTER DESCRIPTION REQUIREMENT:
If a new character is introduced, make sure to give them a thorough physical description. 
This should include: detailed appearance (hair, eyes, body type, distinctive features), clothing, mannerisms, and any unique traits that make them memorable.
The description should be comprehensive enough for readers to clearly visualize the character.
Do not introduce a character without providing these details, even if it takes several sentences.

Ensure vivid descriptions of characters and places.`;
      }

      generationTask = generateText({
        instruction: prompt,
        onChunk: (data) => {
          currentStoryText += data.textChunk;
          storyOutputEl.value = currentStoryText;
          storyOutputEl.scrollTop = storyOutputEl.scrollHeight;
        }
      });

      await generationTask;

    } catch (err) {
      console.error("Generation error:", err);
      if (generationTask !== null) {
        alert("An error occurred while generating the story.");
      }
    } finally {
      generationTask = null;
      generateBtn.textContent = "✨ Generate";
      loadingEl.hidden = true;
      saveState();
    }
  }

  async function analyzeStoryForCharacters(text) {
    if(!text) return 0;
    try {
      let existingCharsInfo = characters.map(c => `Name: ${c.name}, Description: ${c.description}`).join("\n");
      let textToAnalyze = text;

      let prompt = `Analyze the story text snippet below carefully.
      
      TASK: Identify ALL characters mentioned in the text.
      Compare them with the provided list of existing character names and descriptions.

      IMPORTANT: Do your best to identify when an existing character is being refered to in a new/different way, i.e. a character named Abby Boom can be refered to as 'Ms. Boom,' 'Abby,' or by a nickname like 'Abs.'
      
      INSTRUCTIONS:
      1. If a name matches an existing character, provide ONLY the NEW information presented in the present text context that is NOT already in the existing description.
      2. If a name is NEW (not in the existing list, and not refering to a character on the existing list), create a detailed description based on the text context.
      3. Be extremely thorough. Scan for ANY proper nouns or capitalized names that indicate a person, even if mentioned only once.
      4. Return ONLY a raw JSON object.
      
      Existing Characters:
${existingCharsInfo}
      
      OUTPUT JSON FORMAT:
      { 
        "characters": [{"name": "String", "description": "String"}]
      }
      
      IMPORTANT: ONLY return characters that appear in the text snippet below. Do not return characters from the "Existing" list if they are not present in the text.
        -The character descriptions MUST be formatted as a list of traits, where each trait is on a new line starting with a hyphen "-".
        -Example format: 
        -Trait 1
        -Trait 2
        -etc..
        -The character descriptions SHOULD include physical characteristics, nicknames, relationships, and mannerisms. Enduring characteristics only.
        -The character descriptions SHOULD NOT include one-time actions, non-enduring characteristics, nor irrelevant story beats.
        -For existing characters, ONLY return new information not already in their description, formatted in the list style above.
        -If there are not any substantial new pieces of information pertaining to enduring qualities, you can return an empty description for that character.
      
      Story Text to Analyze:
      """${textToAnalyze}"""`;
      
      let response = await generateText(prompt);
      response = response.replace(/```json/g, '').replace(/```/g, '').trim();
      
      let data;
      try {
        data = JSON.parse(response);
      } catch (e) {
        let firstBrace = response.indexOf('{');
        let lastBrace = response.lastIndexOf('}');
        if (firstBrace !== -1 && lastBrace !== -1) {
           response = response.substring(firstBrace, lastBrace + 1);
           data = JSON.parse(response);
        } else {
           throw e;
        }
      }
      
      let updatedCount = 0;
      
      if(data.characters) {
        data.characters.forEach(c => {
          if(!c.name) return;
          let existingCharIndex = characters.findIndex(char => char.name.toLowerCase() === c.name.toLowerCase());
          
          if(existingCharIndex !== -1) {
            let existingDesc = characters[existingCharIndex].description;
            let newDesc = c.description.trim();
            
            if(newDesc && newDesc.length > 0) {
              characters[existingCharIndex].description = existingDesc + "\n" + newDesc;
              updatedCount++;
            }
          } else {
            characters.push(c);
            updatedCount++;
          }
        });
      }
      
      return updatedCount;
    } catch (e) {
      console.error("Character analysis failed:", e);
      return 0;
    }
  }

        -Trait 2
        -Trait 3
        -The setting descriptions SHOULD include physical characteristics, adjacent/connecting spaces, atmosphere, and important features.
        -The setting descriptions SHOULD NOT include one-time events, non-enduring details, nor irrelevant story beats.
        -For existing settings, ONLY return new information not already in their description, formatted in the list style above.
        -If there are not any substantial new pieces of information pertaining to enduring qualities, you can return an empty description for that setting.
      
      Story Text to Analyze:
      """${textToAnalyze}"""`;
      
      let response = await generateText(prompt);
      response = response.replace(/```json/g, '').replace(/```/g, '').trim();
      
      let data;
      try {
        data = JSON.parse(response);
      } catch (e) {
        let firstBrace = response.indexOf('{');
        let lastBrace = response.lastIndexOf('}');
        if (firstBrace !== -1 && lastBrace !== -1) {
           response = response.substring(firstBrace, lastBrace + 1);
           data = JSON.parse(response);
        } else {
           throw e;
        }
      }
      
      let updatedCount = 0;
      
      if(data.settings) {
        data.settings.forEach(s => {
          if(!s.name) return;
          let existingSettingIndex = settings.findIndex(set => set.name.toLowerCase() === s.name.toLowerCase());
          
          if(existingSettingIndex !== -1) {
            let existingDesc = settings[existingSettingIndex].description;
            let newDesc = s.description.trim();
            
            if(newDesc && newDesc.length > 0) {
              settings[existingSettingIndex].description = existingDesc + "\n" + newDesc;
              updatedCount++;
            }
          } else {
            settings.push(s);
            updatedCount++;
          }
        });
      }
      
      return updatedCount;
    } catch (e) {
      console.error("Settings analysis failed:", e);
      return 0;
    }
  }

  window.addCharGenToMemory = function() {
    let text = charGenOutputEl.value.trim();
    if (!text) return alert("Generate or write a character profile first.");
    
    let name = "Unknown Character";
    let desc = text;
    
    let splitIndex = text.indexOf("\n");
    let firstLine = splitIndex > -1 ? text.substring(0, splitIndex).trim() : text.trim();
    
    if (firstLine.toLowerCase().startsWith("name:")) {
      name = firstLine.replace(/^name:\s*/i, "").trim();
    } else {
      name = firstLine;
    }
    
    desc = splitIndex > -1 ? text.substring(splitIndex + 1).trim() : "";
    
    let editingIndex = charGenColumnEl.dataset.editingIndex;
    if (editingIndex !== undefined) {
      characters[editingIndex].name = name;
      characters[editingIndex].description = desc;
      delete charGenColumnEl.dataset.editingIndex;
      renderSidebar();
      saveState();
      alert(`Updated "${name}" in memory!`);
      clearCharGen();
      return;
    }
    
    let existingIndex = characters.findIndex(c => c.name.toLowerCase() === name.toLowerCase());
    if (existingIndex !== -1) {
      if(confirm(`A character named "${name}" already exists. Update their description?`)) {
        characters[existingIndex].description = desc;
        renderSidebar();
        saveState();
        alert(`Updated "${name}" in memory!`);
        clearCharGen();
        return;
      }
    }
    
    characters.push({ name, description: desc });
    renderSidebar();
    saveState();
    
    alert(`Added "${name}" to memory!`);
    clearCharGen();
  };

  function applyStoryData(importedData) {
    characters = importedData.characters || [];
    settings = importedData.settings || [];
    chapterSummaries = importedData.chapterSummaries || [];

    if (importedData.plotAndIdeas) {
        plot = importedData.plotAndIdeas;
        themesStr = importedData.themes || "";
        ideas = importedData.ideas || "";
    } else {
        plot = importedData.plot || "";
        themesStr = importedData.themes || "";
        ideas = importedData.ideas || "";
    }
    
    negativePrompts = importedData.negativePrompts || "";
    
    voice = importedData.voice || "Default";
    bannedWords = importedData.bannedWords || "";
    customSystemPrompt = importedData.customSystemPrompt || "";

    // UI Settings Import - Handle both old and new formats
    if (importedData.customAccentColor && !importedData.selectedTheme && !importedData.customButtonColors) {
      selectedTheme = "default";
      themes.default.accent = importedData.customAccentColor;
      themes.default.accentHover = adjustColorBrightness(importedData.customAccentColor, -20);
    } else {
      selectedTheme = importedData.selectedTheme || "default";
    }

    // Handle Custom Button Colors migration
    if (importedData.customButtonColors) {
      customButtonColors = importedData.customButtonColors;
    } else if (importedData.customButtonColor) {
      customButtonColors = { accent: importedData.customButtonColor, success: "", danger: "", warning: "" };
    } else {
      customButtonColors = { accent: "", success: "", danger: "", warning: "" };
    }

    // Background Theme Import
    if (importedData.selectedBackgroundTheme) {
      selectedBackgroundTheme = importedData.selectedBackgroundTheme;
    } else if (importedData.isLightMode) {
      selectedBackgroundTheme = "light";
    } else {
      selectedBackgroundTheme = "dark";
    }

    // Handle Custom Background Colors migration
    if (importedData.customBgColors) {
      customBgColors = importedData.customBgColors;
    } else if (importedData.customBgColor) {
      customBgColors = { body: importedData.customBgColor, panel: "", input: "" };

      customBgColors = { body: "", panel: "", input: "" };
    }

      selectedFontFamily = importedData.selectedFontFamily;

      // Migrate old font theme to just font family

  window.saveStory = function() {
    if (chapterSummaries.length === 0) {
      alert("No chapters to save yet!");
      return;
    }

    let format = exportFormatSelect.value;
    let content = "";
    let mimeType = "text/plain";
    let fileExtension = "txt";

    if (format === "md") {
      fileExtension = "md";
      mimeType = "text/markdown";
      chapterSummaries.forEach((chapter, index) => {
        content += `# Chapter ${index + 1}: ${chapter.title}\n\n`;
        content += `${chapter.text}\n\n`;
        content += "---\n\n";
      });
    } else if (format === "html") {
      fileExtension = "html";
      mimeType = "text/html";
      content += `<!DOCTYPE html>\n<html lang="en">\n<head>\n<meta charset="UTF-8">\n<title>My Story</title>\n<style>\nbody { font-family: 'Georgia', serif; max-width: 800px; margin: 0 auto; padding: 40px; line-height: 1.7; color: #333; background-color: #fff; }\nh1 { border-bottom: 1px solid #ccc; padding-bottom: 10px; }\nhr { border: 0; border-top: 1px solid #eee; margin: 30px 0; }\n.chapter-content { white-space: pre-wrap; }\n</style>\n</head>\n<body>\n`;
      
      chapterSummaries.forEach((chapter, index) => {
        content += `<h1>Chapter ${index + 1}: ${chapter.title}</h1>\n`;
        let safeText = chapter.text.replace(/&/g, "&amp;").replace(/</g, "&lt;").replace(/>/g, "&gt;");
        content += `<div class="chapter-content">${safeText}</div>\n`;
        content += `<hr>\n`;
      });
      
      content += `</body>\n</html>`;
    } else {
      chapterSummaries.forEach((chapter, index) => {
        content += `CHAPTER ${index + 1}: ${chapter.title}\n\n`;
        content += `${chapter.text}\n\n`;
        content += "----------------------------------------\n\n";
      });
    }
    
    let blob = new Blob([content], { type: mimeType });
    let url = URL.createObjectURL(blob);
    let a = document.createElement("a");
    a.href = url;
    a.download = `story.${fileExtension}`;
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
    URL.revokeObjectURL(url);
  };

  window.saveLocally = async function() {
    try {
      let savedList = await kv.storyGenerator.get("localSavedStories") || [];
      
      if (savedList.length > 0) {
        let overwriteOptions = "Select a story slot to overwrite, or click Cancel to create a new one:\n\n";
        savedList.forEach((item, index) => {
          overwriteOptions += `${index + 1}. ${item.name} (Saved: ${item.date})\n`;
        });
        overwriteOptions += "\nEnter the number of the story to overwrite, or click Cancel to save as new.";
        
        let choice = prompt(overwriteOptions);
        
        if (choice === null) {
          createNewSavedStory(savedList);
          return;
        }
        
        let choiceNum = parseInt(choice);
        if (!isNaN(choiceNum) && choiceNum >= 1 && choiceNum <= savedList.length) {
          await overwriteSavedStory(savedList, savedList[choiceNum - 1]);
        } else if (choice.trim() !== "") {
          createNewSavedStory(savedList, choice.trim());
        } else {
          createNewSavedStory(savedList);
        }
      } else {
        createNewSavedStory(savedList);
      }
    } catch (e) {
      console.error("Failed to save locally:", e);
      alert("Could not save story to local storage.");
    }
  };
  
  async function createNewSavedStory(savedList, providedName) {
    let name = providedName || prompt("Enter a name for this saved story:");
    if (!name) return;
    
    try {
      let fullStoryData = getCurrentStoryData();
      
      let existingIndex = savedList.findIndex(item => item.name.toLowerCase() === name.toLowerCase());
      if (existingIndex !== -1) {
        if (!confirm(`A story named "${name}" already exists. Overwrite it?`)) {
          return;
        }
        savedList[existingIndex] = {
          id: savedList[existingIndex].id,
          name: name,
          data: fullStoryData,
          date: new Date().toLocaleDateString()
        };
      } else {
        savedList.unshift({
          id: Date.now(),
          name: name,
          data: fullStoryData,
          date: new Date().toLocaleDateString()
        });
      }
      
      await kv.storyGenerator.set("localSavedStories", savedList);
      renderSavedStoriesList();
      alert("Story saved locally!");
    } catch (e) {
      console.error("Failed to save story:", e);
      alert("Could not save story.");
    }
  }
  
  async function overwriteSavedStory(savedList, existingStory) {
    if (!confirm(`Overwrite "${existingStory.name}" with current story?`)) {
      return;
    }
    
    try {
      let fullStoryData = getCurrentStoryData();
      
      let index = savedList.findIndex(item => item.id === existingStory.id);
      if (index !== -1) {
        savedList[index] = {
          id: existingStory.id,
          name: existingStory.name,
          data: fullStoryData,
          date: new Date().toLocaleDateString()
        };
      }
      
      await kv.storyGenerator.set("localSavedStories", savedList);
      renderSavedStoriesList();
      alert(`Overwrote "${existingStory.name}"!`);
    } catch (e) {
      console.error("Failed to overwrite story:", e);
      alert("Could not overwrite story.");
    }
  }
  
  window.loadLocally = async function(id) {
    if(!confirm("Are you sure? This will overwrite your current story with the saved version.")) return;
    
    try {
      let savedList = await kv.storyGenerator.get("localSavedStories") || [];
      let item = savedList.find(i => i.id === id);
      
      if (item && item.data) {
        applyStoryData(item.data);
        alert("Story loaded successfully!");
      } else {
        alert("Could not find saved story.");
      }
    } catch (e) {
      console.error("Failed to load locally:", e);
      alert("Could not load story from local storage.");
    }
  };
  
  window.deleteSavedStory = async function(id) {
    if(!confirm("Are you sure you want to delete this saved story?")) return;
    
    try {
      let savedList = await kv.storyGenerator.get("localSavedStories") || [];
      savedList = savedList.filter(i => i.id !== id);
      await kv.storyGenerator.set("localSavedStories", savedList);
      renderSavedStoriesList();
    } catch (e) {
      console.error("Failed to delete story:", e);
      alert("Could not delete story.");
    }
  };
  
  async function renderSavedStoriesList() {
    let container = savedStoriesListEl;
    if(!container) return;
    
    try {
      let savedList = await kv.storyGenerator.get("localSavedStories") || [];
      
      if (savedList.length === 0) {
        container.innerHTML = '<div style="color: var(--text-muted); font-style: italic; text-align: center; padding: 20px;">No locally saved stories found.</div>';
        return;
      }
      
      container.innerHTML = '';
      
      savedList.forEach(item => {
        let div = document.createElement('div');
        div.style.cssText = "background: var(--bg-input); padding: 16px; border-radius: var(--radius-md); border: 1px solid var(--border-color); display: flex; justify-content: space-between; align-items: center; gap: 12px; flex-wrap: wrap;";
        
        div.innerHTML = `
          <div style="flex-grow: 1; min-width: 0;">
            <div style="font-weight: 600; color: var(--text-main); margin-bottom: 6px;">${item.name}</div>
            <div style="font-size: 13px; color: var(--text-muted);">Saved: ${item.date}</div>
          </div>
          <div style="display: flex; gap: 8px; flex-shrink: 0;">
            <button onclick="overwriteFromList(${item.id})" class="btn-warning">Overwrite</button>
            <button onclick="loadLocally(${item.id})" class="btn-success">Load</button>
            <button onclick="deleteSavedStory(${item.id})" class="btn-secondary" style="padding: 8px 12px;">&times;</button>
          </div>
        `;
        
        container.appendChild(div);
      });
    } catch (e) {
      console.error("Failed to render saved stories:", e);
      container.innerHTML = '<div style="color: var(--accent-danger); text-align: center; padding: 20px;">Error loading saved stories.</div>';
    }
    updateRandomGlows();
  }
  
  window.overwriteFromList = async function(id) {
    if(!confirm("Are you sure you want to overwrite this saved story with the current content?")) return;
    
    try {
      let savedList = await kv.storyGenerator.get("localSavedStories") || [];
      let item = savedList.find(i => i.id === id);
      
      if (item) {
        let fullStoryData = getCurrentStoryData();
        
        let index = savedList.findIndex(i => i.id === id);
        if (index !== -1) {
          savedList[index] = {
            id: item.id,
            name: item.name,
            data: fullStoryData,
            date: new Date().toLocaleDateString()
          };
        }
        
        await kv.storyGenerator.set("localSavedStories", savedList);
        renderSavedStoriesList();
        alert(`Overwrote "${item.name}"!`);
      } else {
        alert("Could not find saved story.");
      }
    } catch (e) {
      console.error("Failed to overwrite story:", e);
      alert("Could not overwrite story.");
    }
  }
