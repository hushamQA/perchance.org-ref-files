const UTILS = {

  toOptions: map => map.map(([v, l]) => `<option value="${v}">${l}</option>`).join(''),
  randSeed:  () => Math.floor(Math.random() * 999_999),

  buildPositive(raw, styleKey, genreKey, shapeKey, forceMulti = false) {
    const style = CONFIG.ART_STYLES.find(a => a[0] === styleKey)?.[2] ?? '';
    const genre = CONFIG.GENRES.find(g => g[0] === genreKey)?.[2]     ?? '';
    const hint  = CONFIG.SHAPES[shapeKey]?.hint ?? '';
    const multi = (forceMulti || UTILS.isMulti(raw))
      ? '(distinct separate individuals:1.2),(individual anatomy per figure:1.2),(clear space between each figure:1.1)'
      : '';
    const anat = STATE.anatomy ? CONFIG.ANATOMY_BOOST : '';
    return [raw.trim(), genre, style, hint, '(highly detailed:1.2),(4k:1.1),sharp focus', multi, anat]
      .filter(Boolean).join(', ');
  },

  // استخرج فعل الشخصية المناسب من نص المشهد — extract character action from scene text
  extractAction(scenePrompt, charIdx) {
    for (const [re, actions] of CONFIG.INTERACTION_MAP)
      if (re.test(scenePrompt)) return actions[charIdx % actions.length];
    return null;
  },

  // استخرج توجيه النظر من نص المشهد — extract gaze direction from scene text
  extractGaze(scenePrompt) {
    for (const [re, gaze] of CONFIG.GAZE_MAP)
      if (re.test(scenePrompt)) return `(${gaze}:1.2)`;
    // افتراضي عند عدم وجود حدث — default: avoid direct camera stare
    return '(natural candid gaze, not staring at camera:1.1)';
  },

  // FIX: المشهد أولاً ← الشخصيات كفاعلين ← DNA — scene first, characters as active participants

      return UTILS.buildPositive(rawPrompt, styleKey, genreKey, shapeKey);

    const charParts = [], dnaParts = [];

        hasAction = true;
      } else {
        charParts.push(`(${nm} ${roleLbl}:1.05)`);
      }
      // DNA بوزن أقل — أقواس عادية بدل مربعة لتجنب Perchance list syntax
      dnaParts.push(`(${nm} appearance: ${dna})`);
    });

    const interactionBoost = hasAction
      ? '(characters actively interacting:1.2),(dynamic action pose:1.15),(physically engaged:1.1),(expressive body language:1.1)'
      : '';

    // توجيه النظر حسب الحدث — gaze direction based on scene event
    const gazeDir = UTILS.extractGaze(rawPrompt);

    // ترتيب: المشهد أولاً ← الشخصيات الفاعلة ← DNA ← تعزيز التفاعل ← توجيه النظر

  extractSrc(raw) {
    if (!raw) return null;
    const tag = raw?.tagName?.toUpperCase?.();
    if (tag === 'IMG') return raw.src || null;
    if (tag) { const n = raw.querySelector?.('img'); if (n) return n.src || null; }
    if (raw?.imageId)
      return `https://image-generation.perchance.org/api/downloadTemporaryImage?imageId=${raw.imageId}&fileExtension=${raw.fileExtension ?? 'jpeg'}`;

    }
    if (typeof raw === 'object') {
      for (const k of ['src','url','imageUrl','image_url','href','dataUrl','data_url'])

        r.onloadend = () => res(r.result);
        r.onerror   = rej;
        r.readAsDataURL(blob);
      });
    } catch { return src; }
  },

      if (keys.length >= CONFIG.IMG_LIMIT)
        keys.sort().slice(0, keys.length - CONFIG.IMG_LIMIT + 1).forEach(x => localStorage.removeItem(x));
      localStorage.setItem(k, v);
    } catch(e) { console.warn('STORE.set:', e); }
  },
  histLoad:      () => { try { return JSON.parse(localStorage.getItem(CONFIG.HIST_KEY) ?? '[]'); } catch { return []; } },
  histSave: list => { try { localStorage.setItem(CONFIG.HIST_KEY, JSON.stringify(list)); } catch {} },
  histAdd(entry) {
    const list = STORE.histLoad();
    if (list.some(e => e.key === entry.key)) return;
    list.unshift(entry);
    if (list.length > CONFIG.IMG_LIMIT) STORE.del(list.splice(CONFIG.IMG_LIMIT)[0]?.key);
    STORE.histSave(list);
  },
  histDel: key => { STORE.del(key); STORE.histSave(STORE.histLoad().filter(e => e.key !== key)); },
};

// CHAR_STORE

const CHAR_STORE = {
  KEY:     'clib_chars',
  IMG_PFX: 'clib_img_',

  list()   { try { return JSON.parse(localStorage.getItem(CHAR_STORE.KEY) ?? '[]'); } catch { return []; } },
  save(ch) { try { localStorage.setItem(CHAR_STORE.KEY, JSON.stringify(ch)); } catch(e) { console.warn(e); } },
  get(id)  { return CHAR_STORE.list().find(c => c.id === id) ?? null; },

  upsert(char) {
    const list = CHAR_STORE.list();
    const idx  = list.findIndex(c => c.id === char.id);
    if (idx >= 0) list[idx] = char; else list.push(char);
    CHAR_STORE.save(list);
  },

  delete(id) {
    const char = CHAR_STORE.get(id);
    for (const k of [char?.portraitKey, char?.fullbodyKey])
      if (k) { try { localStorage.removeItem(k); } catch {} }
    CHAR_STORE.save(CHAR_STORE.list().filter(c => c.id !== id));
    STATE.cast = STATE.cast.filter(e => e.charId !== id);
  },

  setPortrait(charId, dataUrl) {
    const key  = `${CHAR_STORE.IMG_PFX}${charId}_bust`;
    try { localStorage.setItem(key, dataUrl); } catch(e) { console.warn(e); }
    const char = CHAR_STORE.get(charId);
    if (char) { char.portraitKey = key; CHAR_STORE.upsert(char); }
    return key;
  },

  getPortrait(charId) {
    const char = CHAR_STORE.get(charId);
    if (!char?.portraitKey) return null;
    try { return localStorage.getItem(char.portraitKey); } catch { return null; }
  },

  setFullbody(charId, dataUrl) {
    const key  = `${CHAR_STORE.IMG_PFX}${charId}_full`;
    try { localStorage.setItem(key, dataUrl); } catch(e) { console.warn(e); }
    const char = CHAR_STORE.get(charId);
    if (char) { char.fullbodyKey = key; CHAR_STORE.upsert(char); }
    return key;
  },

  getFullbody(charId) {
    const char = CHAR_STORE.get(charId);
    if (!char?.fullbodyKey) return null;
    try { return localStorage.getItem(char.fullbodyKey); } catch { return null; }
  },
};

const CHAR = {
  buildDNA(data) {
    const tok = (map, val) => map.find(x => x[0] === val)?.[2] ?? '';
    const pts = [];

    const age    = data.age ? `${data.age} year old` : '';
    const gender = tok(CHAR_DNA.GENDER, data.gender);
    if (age || gender) pts.push([age, gender].filter(Boolean).join(' '));

    const et = tok(CHAR_DNA.ETHNICITY, data.ethnicity);  if (et) pts.push(et);
    const sk = tok(CHAR_DNA.SKIN,      data.skin);        if (sk) pts.push(sk);
    const st = tok(CHAR_DNA.SKIN_TEX,  data.skinTex);     if (st) pts.push(st);
    const fs = tok(CHAR_DNA.FACE_SHAPE,data.faceShape);   if (fs) pts.push(fs);

    const eyeC = tok(CHAR_DNA.EYE_COLOR, data.eyeColor);
    const eyeS = tok(CHAR_DNA.EYE_SHAPE, data.eyeShape);
    if (eyeC || eyeS) pts.push([eyeS, eyeC].filter(Boolean).join(' '));

    const hC = tok(CHAR_DNA.HAIR_COLOR,  data.hairColor);
    const hL = tok(CHAR_DNA.HAIR_LENGTH, data.hairLength);
    const hS = tok(CHAR_DNA.HAIR_STYLE,  data.hairStyle);
    if (hC || hL || hS) pts.push([hS, hL, hC].filter(Boolean).join(' '));

    const br = tok(CHAR_DNA.EYEBROWS, data.eyebrows); if (br) pts.push(br);
    const no = tok(CHAR_DNA.NOSE,     data.nose);      if (no) pts.push(no);
    const lp = tok(CHAR_DNA.LIPS,     data.lips);      if (lp) pts.push(lp);
    const ht = tok(CHAR_DNA.HEIGHT,   data.height);    if (ht) pts.push(ht);
    const bd = tok(CHAR_DNA.BUILD,    data.build);     if (bd) pts.push(bd);

    (data.marks       ?? []).forEach(m => { const t = tok(CHAR_DNA.MARKS,       m); if (t) pts.push(t); });
    (data.accessories ?? []).forEach(a => { const t = tok(CHAR_DNA.ACCESSORIES, a); if (t) pts.push(t); });

    if (data.extra?.trim()) pts.push(data.extra.trim());

    return pts.join(', ');
  },

  },

const LOGIC = {
  // FIX: generateText الصحيح لـ Perchance — correct Perchance ai-text-plugin API
  async callAI(instruction, userContent) {
    if (typeof generateText === 'undefined')
      throw new Error('ai-text-plugin not loaded in List Editor.');
    let result = '';
    const full = userContent?.trim()
      ? `${instruction}\n\n---\n${userContent}`
      : instruction;
    const obj = generateText({
      instruction: full,
      onChunk: c => { result += typeof c === 'string' ? c : (c?.chunk ?? c?.text ?? c?.textChunk ?? ''); },
    });
    if      (obj instanceof Promise) await obj;
    else if (obj?.onFinishPromise)   await obj.onFinishPromise;
    return result.trim();
  },

  async genImage(positive, negative, seed, w, h) {
    if (typeof image === 'undefined')
      throw new Error('text-to-image-plugin not loaded in List Editor.');
    const s   = seed ?? UTILS.randSeed();
    const raw = await image({ prompt: positive, negativePrompt: negative, seed: s, width: w, height: h });
    const src = UTILS.extractSrc(raw);
    if (!src) throw new Error('Plugin returned no usable image source.');
    const dataUrl = await UTILS.toDataUrl(src);
    if (!dataUrl) throw new Error('Could not convert image to dataURL.');
    return { dataUrl, seed: s };
  },

  buildEnhanceInstruction(styleKey, genreKey, isMulti) {
    const style = CONFIG.ART_STYLES.find(a => a[0] === styleKey)?.[1] ?? 'Default';
    const genre = CONFIG.GENRES.find(g => g[0] === genreKey)?.[1]     ?? '';
    return [
      'You are an expert image prompt engineer for AI image generators.',
      `Art Style: ${style}`,
      genre   && `Genre/Theme: ${genre}`,
      isMulti && 'IMPORTANT: Multiple people — keep count explicit, describe each figure separately.',
      'Rules: English only, max 60 words. VISUAL elements only: lighting, composition, colors, textures. Output ONLY the improved prompt — no preamble, no quotes.',
    ].filter(Boolean).join('\n');
  },

  // FIX: instruction منسجم مع العرق + يحترم المدخلات اليدوية

  
  // توليد عشوائي فوري من CHAR_DNA بدون AI — instant randomize from CHAR_DNA, zero AI calls
  randomizeCharLocal(sections, currentDNA) {
    const rand    = arr => arr[Math.floor(Math.random() * arr.length)];
    const randN   = (min, max) => Math.floor(Math.random() * (max - min + 1)) + min;
    const randArr = (arr, max) => [...arr].sort(() => Math.random() - .5).slice(0, randN(0, max));

    const SEC_FIELDS = {
      identity: ['gender','age','ethnicity','skin','skinTex'],
      face:     ['faceShape','eyeColor','eyeShape','hairColor','hairLength','hairStyle','eyebrows','nose','lips','marks','accessories'],
      body:     ['height','build'],
    };

    const ethnicity = currentDNA.ethnicity || '';
    const ethCtx    = CONFIG.ETHNICITY_CONTEXT[ethnicity];
    const result    = {};

    sections.forEach(s => (SEC_FIELDS[s] ?? []).forEach(k => {
      // احترام المدخلات اليدوية إن كان مفعّلاً — respect manual inputs if enabled
      if (STATE.respectManual) {
        const v = currentDNA[k];
        if (Array.isArray(v) ? v.length > 0 : (v && v !== '')) return;
      }

      // خريطة الحقل → القائمة المناسبة — field → pool mapping
      const POOLS = {
        gender:     CHAR_DNA.GENDER.map(x=>x[0]).filter(Boolean),
        ethnicity:  CHAR_DNA.ETHNICITY.slice(1).map(x=>x[0]),
        skin:       CHAR_DNA.SKIN.slice(1).map(x=>x[0]),
        skinTex:    CHAR_DNA.SKIN_TEX.slice(1).map(x=>x[0]),
        faceShape:  CHAR_DNA.FACE_SHAPE.slice(1).map(x=>x[0]),
        eyeShape:   CHAR_DNA.EYE_SHAPE.slice(1).map(x=>x[0]),
        hairLength: CHAR_DNA.HAIR_LENGTH.slice(1).map(x=>x[0]),
        hairStyle:  CHAR_DNA.HAIR_STYLE.slice(1).map(x=>x[0]),
        eyebrows:   CHAR_DNA.EYEBROWS.slice(1).map(x=>x[0]),
        nose:       CHAR_DNA.NOSE.slice(1).map(x=>x[0]),
        lips:       CHAR_DNA.LIPS.slice(1).map(x=>x[0]),
        height:     CHAR_DNA.HEIGHT.slice(1).map(x=>x[0]),
        build:      CHAR_DNA.BUILD.slice(1).map(x=>x[0]),
        // قيود العرق لأهم الحقول — ethnicity-constrained pools
        eyeColor:   ethCtx?.eyeColor  ?? CHAR_DNA.EYE_COLOR.slice(1).map(x=>x[0]),
        hairColor:  ethCtx?.hairColor ?? CHAR_DNA.HAIR_COLOR.slice(1).map(x=>x[0]),
      };

      if (k === 'age')         result.age         = randN(1, 55);
      else if (k === 'marks')  result.marks        = randArr(CHAR_DNA.MARKS.map(x=>x[0]), 3);
      else if (k === 'accessories') result.accessories = randArr(CHAR_DNA.ACCESSORIES.map(x=>x[0]), 2);
      else if (POOLS[k])       result[k]           = rand(POOLS[k]);
    }));

    if (!Object.keys(result).length)
      throw new Error('All fields already set — toggle 🔓 Overwrite All to randomize filled fields.');

    return result;
  },
};

const App = {
  readInputs: () => ({
    rawPrompt: EL.promptIn.value.trim(),
    styleKey:  EL.styleSel.value,
    genreKey:  EL.genreSel.value,
    shapeKey:  EL.shapeSel.value,
    userNeg:   EL.negIn.value,
    count:     parseInt(EL.countSel.value) || 6,
    isRandom:  EL.randSeed.checked,
    seedVal:   parseInt(EL.seedIn.value) || null,
  }),

  async generate(opts = {}) {
    if (STATE.busy) return;
    const { posOverride=null, negOverride=null, countOverride=null, seedOverride=null, targetCard=null } = opts;
    const inp = App.readInputs();
    if (!inp.rawPrompt && !posOverride) { EL.promptIn.focus(); return; }

    const shape    = CONFIG.SHAPES[inp.shapeKey] ?? CONFIG.SHAPES.square;
    const positive = posOverride ?? UTILS.buildCastPositive(inp.rawPrompt, inp.styleKey, inp.genreKey, inp.shapeKey);
    const negative = negOverride ?? UTILS.buildNegative(inp.rawPrompt, inp.userNeg);
    const count    = countOverride ?? inp.count;
    // استخدام seed الشخصية الأولى لضمان الثبات — use first cast char seed for consistency
    const castSeed = STATE.cast.length ? (CHAR_STORE.get(STATE.cast[0].charId)?.seed ?? null) : null;
    const seed     = seedOverride ?? (inp.isRandom ? castSeed : inp.seedVal);

    STATE.busy = true; EL.btnGen.disabled = true;
    try {
      if (targetCard) {
        UI.setStatus('Regenerating…', 'busy');
        await App._genCard(targetCard, positive, negative, seed, shape, inp);
        UI.setStatus('Done ✓', 'ready'); UI.renderHistory();
      } else {
        UI.setupGrid(count, shape);
        UI.setStatus(`Generating 0/${count}…`, 'busy');
        const cards = [...EL.batchGrid.querySelectorAll('.batch-card')];
        let done = 0;
        // توليد متوازٍ بدفعات من 3 — parallel generation in batches of 3
        const BATCH = 3;
        for (let i = 0; i < cards.length; i += BATCH) {
          const batch = cards.slice(i, i + BATCH);
          await Promise.all(batch.map((card, j) =>
            App._genCard(card, positive, negative, (i === 0 && j === 0) ? seed : null, shape, inp)
              .then(() => UI.setStatus(`Generated ${++done}/${count}…`, 'busy'))
          ));
        }
        UI.setStatus(`Done ✓ · ${done}/${count}`, 'ready'); UI.renderHistory();
      }
    } finally { STATE.busy = false; EL.btnGen.disabled = false; }
  },

  async _genCard(card, positive, negative, seed, shape, inp) {
    try {
      const { dataUrl, seed:s } = await LOGIC.genImage(positive, negative, seed, shape.w, shape.h);
      const key = UTILS.genKey();
      STORE.set(key, dataUrl);
      Object.assign(card.dataset, { key, seed:s, pos:positive, neg:negative, raw:inp?.rawPrompt??'', style:inp?.styleKey??'', genre:inp?.genreKey??'', shape:inp?.shapeKey??'square' });
      UI.fillCard(card, key);
      STORE.histAdd({ key, seed:s, prompt:inp?.rawPrompt??'', styleKey:inp?.styleKey??'', genreKey:inp?.genreKey??'', shapeKey:inp?.shapeKey??'square' });
    } catch(err) { UI.showCardError(card, err.message); }
  },

  async enhance() {
    const raw = EL.promptIn.value.trim();
    if (!raw || STATE.busy) return;
    STATE.busy = true; EL.btnEnhance.disabled = true; EL.btnEnhance.textContent = '⏳';
    UI.setStatus('Enhancing…', 'busy');
    try {
      const enhanced = await LOGIC.callAI(
        LOGIC.buildEnhanceInstruction(EL.styleSel.value, EL.genreSel.value, UTILS.isMulti(raw)), raw);
      EL.promptIn.value = enhanced;
      UI.updateMultiBadge(enhanced);

    } catch(err) { UI.setStatus('Enhance failed: ' + err.message, 'err');
    } finally { STATE.busy = false; EL.btnEnhance.disabled = false; EL.btnEnhance.textContent = '✨ AI Enhance'; }
  },

  handleCorner(btn) {

      const key = UTILS.genKey(); STORE.set(key, dataUrl);
      Object.assign(card.dataset, { key, seed:s, pos:positive, neg:negative });
      UI.fillCard(card, key);
      STORE.histAdd({ key, seed:s, prompt:card.dataset.raw??'', styleKey:card.dataset.style??'', genreKey:card.dataset.genre??'', shapeKey:card.dataset.shape??'square' });
      UI.renderHistory(); UI.setStatus('Regenerated ✓', 'ready');
    } catch(err) { UI.showCardError(card, err.message); UI.setStatus('Error: ' + err.message, 'err');
    } finally { STATE.busy = false; EL.btnGen.disabled = false; }
  },

  handlePopupAction(action, dataset) {

    if (action === 'hist') {
      UI.hidePopup();

        STORE.histAdd({ key:card.dataset.key, seed:card.dataset.seed, prompt:card.dataset.raw??'', styleKey:card.dataset.style??'', genreKey:card.dataset.genre??'', shapeKey:card.dataset.shape??'square' });
        UI.renderHistory(); UI.setStatus('Saved to history ✓', 'ready');
      }
    } else if (action === 'dl') {
      UI.hidePopup();
      const src = STORE.get(card?.dataset?.key); if (!src) return;
      Object.assign(document.createElement('a'), { href:src, download:`imagegen_${card.dataset.seed ?? Date.now()}.jpg` }).click();
    } else if (action === 'regen-edited') {
      App.regenWithEdited(card);
    } else if (action === 'copy-dna') {
      const dna = decodeURIComponent(dataset.dna ?? '');
      navigator.clipboard?.writeText(dna).then(() => UI.setStatus('DNA copied ✓', 'ready'));
      UI.hidePopup();
    } else if (action === 'close') { UI.hidePopup(); }
  },

  handleFlag(reason) {

  },

    EL.setHd .addEventListener('click', () => UI.toggleSection(EL.setBody,  EL.setArr));
    EL.negHd .addEventListener('click', () => UI.toggleSection(EL.negBody,  EL.negArr));
    EL.castHd.addEventListener('click', () => UI.toggleSection(EL.castBody, EL.castArr));

    EL.btnAddCast.addEventListener('click', () => LIB_APP.addCast());
    EL.btnNewChar .addEventListener('click', () => LIB_APP.openNew());
    EL.btnBack    .addEventListener('click', () => LIB_APP.backToGrid());
    EL.btnSaveChar.addEventListener('click', () => LIB_APP.saveChar());
    EL.btnGenBoth .addEventListener('click', () => LIB_APP.genBothPortraits('both'));
    EL.btnRandAll .addEventListener('click', () => LIB_APP.randSection(['identity','face','body']));
    EL.btnNewSeed .addEventListener('click', () => { EL.edSeed.value = UTILS.randSeed(); });
    EL.btnCopyDNA .addEventListener('click', () => {
      const dna = document.getElementById('dna-preview')?.textContent ?? '';
      navigator.clipboard?.writeText(dna).then(() => UI.setStatus('DNA copied ✓', 'ready'));
    });

    // زر toggle احترام المدخلات اليدوية — respect manual toggle
    EL.btnClearForm.addEventListener('click', () => LIB_APP.clearForm());

    EL.btnNewSeed2.addEventListener('click', () => {
      const v = UTILS.randSeed();
      EL.edSeed.value  = v;
      EL.edSeed2.value = v;
    });

    EL.edSeed .addEventListener('input', () => { EL.edSeed2.value = EL.edSeed.value; });
    EL.edSeed2.addEventListener('input', () => { EL.edSeed.value  = EL.edSeed2.value; });

    EL.btnRespectToggle.addEventListener('click', () => {
      STATE.respectManual = !STATE.respectManual;
      EL.btnRespectToggle.dataset.state = STATE.respectManual ? 'on' : 'off';
      EL.btnRespectToggle.textContent   = STATE.respectManual ? '🔒 Respect Manual' : '🔓 Overwrite All';
    });

    EL.promptIn.addEventListener('keydown', e => { if ((e.ctrlKey || e.metaKey) && e.key === 'Enter') { e.preventDefault(); App.generate(); } });

      // إغلاق عند النقر على الخلفية فقط — close only on backdrop click
      if (e.target === EL.lightbox) UI.closeLightbox();
    });

    EL.lbNext.addEventListener('click', e => { e.stopPropagation(); UI.lbNav(+1); });

    // تنقل بلوحة المفاتيح — keyboard navigation
    document.addEventListener('keydown', e => {
      if (EL.lightbox.classList.contains('hidden')) return;
      if (e.key === 'ArrowLeft')  { e.preventDefault(); UI.lbNav(-1); }
      if (e.key === 'ArrowRight') { e.preventDefault(); UI.lbNav(+1); }
    });

    document.querySelector('.tab-bar').addEventListener('click', e => {
      const tab = e.target.closest('.tab'); if (tab) UI.showTab(tab.dataset.tab);
    });

    EL.batchGrid.addEventListener('click', e => {
      const btn = e.target.closest('.cbtn'); if (btn) { e.stopPropagation(); App.handleCorner(btn); }
    });

    EL.popup.addEventListener('click', e => {
      e.stopPropagation();
      if (e.target.closest('.pop-close')) { UI.hidePopup(); return; }
      const pa = e.target.closest('[data-pa]'); if (pa) { App.handlePopupAction(pa.dataset.pa, pa.dataset); return; }
      const pf = e.target.closest('[data-pf]'); if (pf) { App.handleFlag(pf.dataset.pf); return; }
    });

    EL.histGrid.addEventListener('click', e => {
      const del  = e.target.closest('[data-del]');
      const card = e.target.closest('.hist-card');
      if (del)  { e.stopPropagation(); STORE.histDel(del.dataset.del); UI.renderHistory(); return; }
      if (card) UI.openLightbox(card.dataset.key);
    });

    EL.castList.addEventListener('click', e => {
      const del = e.target.closest('[data-cast-del]');
      if (del) { STATE.cast.splice(parseInt(del.dataset.castDel), 1); LIB_UI.renderCastList(); UI.updateCastPanel(); }
    });
    EL.castList.addEventListener('change', e => {
      const cc = e.target.dataset.castChar;
      const cr = e.target.dataset.castRole;
      if (cc !== undefined) { STATE.cast[parseInt(cc)].charId = e.target.value; LIB_UI.renderCastList(); }
      if (cr !== undefined)   STATE.cast[parseInt(cr)].role   = e.target.value;
    });

    EL.libGrid.addEventListener('click', e => {
      const cc   = e.target.closest('[data-cc]');
      const edit = e.target.closest('.btn-edit');
      if (cc) {
        e.stopPropagation();
        const charId = cc.closest('.char-card')?.dataset.charId; if (!charId) return;
        STATE.activeCard = cc.closest('.char-card');
        LIB_APP.handleCharCorner(cc.dataset.cc, charId, cc);
      } else if (edit) {
        const charId = edit.closest('.char-wrap')?.querySelector('.char-card')?.dataset.charId; if (!charId) return;
        LIB_APP.openEditor(charId);
      }
    });

    document.querySelector('.ed-portrait').addEventListener('click', e => {
      const ecc = e.target.closest('[data-ecc]'); if (!ecc) return;
      e.stopPropagation(); LIB_APP.handleEditorCorner(ecc.dataset.ecc);
    });
    EL.epImgBust.addEventListener('click', () => {
      const ch = CHAR_STORE.get(STATE.editingChar); if (ch?.portraitKey) UI.openLightbox(ch.portraitKey);
    });
    EL.epImgFull.addEventListener('click', () => {
      const ch = CHAR_STORE.get(STATE.editingChar); if (ch?.fullbodyKey) UI.openLightbox(ch.fullbodyKey);
    });

    document.querySelectorAll('.sec-hd').forEach(hd => {
      hd.addEventListener('click', e => {
        if (e.target.closest('.btn-rand')) return; // تجاهل زر 🎲 — ignore rand button
        const body = document.getElementById(`sec-${hd.dataset.sec}`);
        const arr  = hd.querySelector('.arr');
        if (body) UI.toggleSection(body, arr);
      });
    });

    document.getElementById('ed-form').addEventListener('change', () => LIB_UI.updateDNAPreview());
    document.getElementById('ed-form').addEventListener('click', e => {
      const tag  = e.target.closest('.tag');
      const rsec = e.target.closest('[data-rand-sec]');
      if (tag)  { tag.classList.toggle('on'); LIB_UI.updateDNAPreview(); return; }
      if (rsec) { e.stopPropagation(); LIB_APP.randSection([rsec.dataset.randSec]); }
    });

    document.addEventListener('click', e => {
      if (!e.target.closest('#g-popup') && !e.target.closest('.cbtn') && !e.target.closest('[data-cc]'))
        UI.hidePopup();
    });
    document.addEventListener('keydown', e => {
      if (e.key === 'Escape') { UI.closeLightbox(); UI.hidePopup(); }
    });
  },
};
